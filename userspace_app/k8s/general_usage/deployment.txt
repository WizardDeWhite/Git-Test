1. Deployment Yaml Example

```
apiVersion: apps/v1
kind: Deployment
metadata:
    name: nginx-deployment
    labels:
        app: nginx
spec:
    replicas: 3
    selector:
        matchLabels:
            app: nginx
    template:
        metadata:
            labels:
                app: nginx
        spec:
            containers:
            - name: nginx
              env:
              - name: CLUSTER_NM
                value: "test"
              image: nginx:latest
              ports:
              - containerPort: 80
```
